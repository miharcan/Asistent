    JAVA EXAMPLE:

    import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.net.MalformedURLException;
import java.net.URL;
import java.net.URLConnection;
import java.net.URLEncoder;

public class RestApiExample {
	static String endpoint = "http://server1.nlp.insight-centre.org/cgi-bin/asistent_restAPI.cgi?json=";
	@SuppressWarnings("deprecation")
	public static String callApi(){
		String extractText = "{\"method\":\"phrase_based\",\"key\":\"\",\"translation_direction\":\"en_sl\",\"nbest\":\"20\",\"text2translate\":[{\"source\":\"This is a sentence\"},{\"source\":\"The Weather is nice\"},{\"source\":\"Thank God it is Friday\"},{\"source\":\"I am working on Sunday\"}]}";
		extractText = URLEncoder.encode(extractText).replace("+", "%20");
		String query = endpoint + extractText;
		StringBuffer buffer = new StringBuffer();
		try {
			URL url = new URL(query);
			URLConnection connection = url.openConnection();
			BufferedReader reader = new BufferedReader(new InputStreamReader(connection.getInputStream()));			
			String inputLine = null;
			while ((inputLine = reader.readLine()) != null){ 
				buffer.append(inputLine);
			}
			reader.close();			
		} catch (MalformedURLException e) {
			e.printStackTrace();
		} catch (IOException e) {
			e.printStackTrace();
		}
		return buffer.toString();
	}
	public static void main(String[] args) {
		String outJson = callApi();		
		System.out.println(outJson);
	}
}

    PYTHON EXAMPLE:

    import urllib
import requests
jsonRequest = '{"method":"phrase_based","key":"","translation_direction":"en_sl","nbest":"20","text2translate":[{"source":"This is a sentence"},{"source":"The Weather is nice"},{"source":"Thank God it is Friday"},{"source":"I am working on Sunday"}]}'
url ="http://server1.nlp.insight-centre.org/cgi-bin/asistent_restAPI.cgi?json="
data =urllib.quote(jsonRequest)
r = requests.get(url+data)
translations = r.json()
print( translations )
